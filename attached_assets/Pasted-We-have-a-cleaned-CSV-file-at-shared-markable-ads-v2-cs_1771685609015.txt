We have a cleaned CSV file at shared/markable-ads-v2.csv with 525 valid ad rows across 81 creators. Replace all hardcoded creator data with CSV-parsed data. No database needed — parse once on load and hold in memory.
The CSV has these exact column names:

Meta Library ID — ad library ID (strip "Library ID: " prefix)
Started Date — start date (strip "Started running on " prefix)
Profile Image — creator profile photo URL
Influencer Facebook Page — creator Facebook page URL (use the slug after the last / as the creator ID)
Ad Details — ad copy text
Video URL — video asset URL (empty for static ads)
Content Image URL — static image URL (empty for video ads)
CTA Shop Now URL — affiliate link
Ad Company — ignore this column

Only include rows where Ad Details is non-empty AND at least one of Video URL or Content Image URL is non-empty. Skip all other rows.
Install papaparse if not already installed: npm install papaparse

function buildCreatorsFromCSV(rows) {
  const map = {};

  rows.forEach(row => {
    const hasCopy = row['Ad Details']?.trim();
    const hasAsset = row['Video URL']?.trim() || row['Content Image URL']?.trim();
    if (!hasCopy || !hasAsset) return;

    const fbUrl = row['Influencer Facebook Page']?.trim();
    const id = fbUrl ? fbUrl.replace(/\/$/, '').split('/').pop().toLowerCase() : null;
    if (!id) return;

    if (!map[id]) {
      map[id] = {
        id,
        name: formatCreatorName(id),
        handle: `@${id}`,
        niche: inferNiche(id, row['Ad Details']),
        tone: 'Authentic, relatable',
        audience: 'Women 25–45',
        color: assignColor(id),
        emoji: assignEmoji(id, row['Ad Details']),
        profileImage: row['Profile Image']?.trim() || null,
        facebookPage: fbUrl,
        existingAds: [],
        products: [],
      };
    }

    const hasVideo = !!row['Video URL']?.trim();
    const hasStatic = !!row['Content Image URL']?.trim();

    map[id].existingAds.push({
      started: (row['Started Date'] || '').replace('Started running on ', '').trim(),
      copy: row['Ad Details'].trim(),
      hasVideo,
      hasStatic,
      videoUrl: row['Video URL']?.trim() || null,
      imageUrl: row['Content Image URL']?.trim() || null,
      shopUrl: row['CTA Shop Now URL']?.trim() || null,
      libraryId: (row['Meta Library ID'] || '').replace('Library ID: ', '').trim(),
    });
  });

  return Object.values(map)
    .map(c => {
      c.totalAds = c.existingAds.length;
      const hasV = c.existingAds.some(a => a.hasVideo);
      const hasS = c.existingAds.some(a => a.hasStatic);
      c.adFormat = hasV && hasS ? 'Video + Static' : hasV ? 'Video Only' : 'Static Only';
      c.products = inferProducts(c.niche);
      return c;
    })
    .sort((a, b) => b.totalAds - a.totalAds);
}

    'andreajeanaco': 'Andrea Jean Co',
  };
  if (overrides[slug.toLowerCase()]) return overrides[slug.toLowerCase()];
  // Auto-format slug into readable name
  return slug
    .replace(/([a-z])([A-Z])/g, '$1 $2')
    .replace(/[_\-.]/g, ' ')
    .replace(/\d+$/, '')
    .split(' ')
    .map(w => w.charAt(0).toUpperCase() + w.slice(1))
    .join(' ')
    .trim();
}
Step 2 — Add helper functions
javascriptconst COLORS = ['#C084FC','#FB923C','#34D399','#60A5FA','#F472B6','#A78BFA','#FCD34D','#F87171'];

function assignColor(id) {
  let hash = 0;
  for (let c of id) hash = (hash * 31 + c.charCodeAt(0)) % COLORS.length;
  return COLORS[hash];
}

function assignEmoji(id, copy = '') {
  const text = (id + ' ' + copy).toLowerCase();
  if (text.includes('fashion') || text.includes('style') || text.includes('outfit') || text.includes('chic')) return '✨';
  if (text.includes('home') || text.includes('decor') || text.includes('interior') || text.includes('farmhouse')) return '🏡';
  if (text.includes('mom') || text.includes('kids') || text.includes('children') || text.includes('school')) return '👶';
  if (text.includes('food') || text.includes('snack') || text.includes('recipe') || text.includes('kitchen')) return '🍽️';
  if (text.includes('travel') || text.includes('vacation') || text.includes('planes')) return '✈️';
  if (text.includes('fitness') || text.includes('workout') || text.includes('curves')) return '💪';
  if (text.includes('planner') || text.includes('productivity') || text.includes('remarkable')) return '📋';
  if (text.includes('beauty') || text.includes('makeup') || text.includes('lipstick')) return '💄';
  return '🌿';
}

function inferNiche(id, copy = '') {
  const text = (id + ' ' + copy).toLowerCase();
  if (text.includes('fashion') || text.includes('style') || text.includes('outfit') || text.includes('chic') || text.includes('closet')) return 'Fashion & Lifestyle';
  if (text.includes('home') || text.includes('decor') || text.includes('interior') || text.includes('farmhouse') || text.includes('renovation')) return 'Home & Decor';
  if (text.includes('mom') || text.includes('kids') || text.includes('children') || text.includes('school') || text.includes('baby')) return 'Mom & Kids';
  if (text.includes('food') || text.includes('snack') || text.includes('recipe') || text.includes('kitchen') || text.includes('foodie')) return 'Food & Kitchen';
  if (text.includes('travel') || text.includes('vacation') || text.includes('planes')) return 'Travel & Lifestyle';
  if (text.includes('fitness') || text.includes('workout') || text.includes('curves') || text.includes('activewear')) return 'Fitness & Wellness';
  if (text.includes('planner') || text.includes('productivity') || text.includes('remarkable')) return 'Productivity';
  if (text.includes('beauty') || text.includes('makeup') || text.includes('lipstick') || text.includes('glam')) return 'Beauty & Fashion';
  return 'Lifestyle & Finds';
}
